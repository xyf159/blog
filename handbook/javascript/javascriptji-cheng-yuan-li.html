    # JavaScript继承原理

    • # 原型链继承

        function Dad() {
        this.name = 'dad'
        this.number = [1, 2, 3]
    }

    function Son() {
        this.name2 = 'son'
    }
    Son.prototype = new Dad()
    let s1 = new Son()
    let s2 = new Son()
    s1.number.push(4)
    console.log(s1.number, s2.number); //(4) [1, 2, 3, 4] (4) [1, 2, 3, 4]
    //因为两个实例使用的是同一个原型对象，内存空间是共享的

    • # 构造函数继承

    function Dad() {
        this.name = 'dad'
        this.number = [1, 2, 3]
        function getName() {
            console.log('hello Dad');
        }
    }
    function Son() {
        Dad.call(this)
        this.name2 = 'son'
    }
    let s3 = new Son1()
    console.log(s3.name);   // dad
    console.log(s3.getName()); //Uncaught TypeError: s3.getName is not a function

    • # 组合继承

        function Dad() {
        this.name = 'dad'
        this.number = [1, 2, 3]
        this.getName = function() {
            console.log('hello Dad');
        }
    }

    function Son() {
        Dad.call(this)
        this.name2 = 'son'
    }
    Son.prototype = new Dad()
    let s1 = new Son()
    let s2 = new Son()
    s1.number.push(4)
    console.log(s1.number, s2.number); //(4) [1, 2, 3, 4] (3) [1, 2, 3]
    s1.getName(); // hello Dad

    • # 原型式继承

     let dad = {
        name: "parent4",
        friends: ["p1", "p2", "p3"],
        getName: function() {
            return this.name;
        }
    };
    let son = Object.create(dad)
    son.name = 'mySon'
    son.friends.push('p4')

    let son1 = Object.create(dad)
    son.friends.push('p5')
    console.log(son.name, son1.name); //mySon parent4
    console.log(son.friends); // ["p1", "p2", "p3","p4","p5"]
    console.log(son1.friends); //["p1", "p2", "p3","p4","p5"]

    • # 寄生式继承

       let dad = {
            name: "parent4",
            friends: ["p1", "p2", "p3"],
            getName: function() {
                return this.name;
            }
        };

        function clone(orginal) {
            let clone = Object.create(orginal)
            clone.getFriend = function() {
                return this.friends
            }
            return clone
        }
        let son = clone(dad)
        console.log(son.getName()); //parent4
        console.log(son.getFriend()); // ["p1", "p2", "p3"]

    • # 寄生组合式继承

        function clone(son, dad) {
        son.prototype = Object.create(dad.prototype)
        son.prototype.constructor = son
    }

    function Dad1() {
        this.name = 'myDad'
        this.number = [1, 2, 3]
    }
    Dad1.prototype.getName = function() {
        return this.name
    }

    function Son1() {
        Dad1.call(this)
        this.friends = 'child5'
    }
    clone(Son1, Dad1)
    Son1.prototype.getFriends = function() {
        return this.friends
    }
    let myson = new Son1()
    console.log(myson.name); //myDad
    console.log(myson.friends); //child5
    console.log(myson.getName()); //myDad
    console.log(myson.getFriends()); //child5

    • # extends实现继承，其原理如上寄生组合式继承

        class Dad {
        constructor(name) {
            this.name = name
        }
        getName() {
            console.log('Person : ' + this.name);
        }
    }
    class Son extends Dad {
        constructor(name, friends) {
            super(name)
            this.friends = friends
        }
    }
    const son1 = new Son('myson', 'myfriends')
    son1.getName() // Person : myson
